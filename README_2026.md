```javascript
// دالة البحث المخصصة لهذا الإصدار
function searchInScopus(searchTerm) {
    // الكود سيتكيف تلقائياً مع هيكل البيانات الحالي
    const mainSheet = findMainSheet();
    if (!mainSheet) return [];
    
    return mainSheet.filter(record => {
        const title = record["Source Title"] || record["Title"] || record["Journal"] || "";
        const issn = record["ISSN"] || record["EISSN"] || "";
        return title.includes(searchTerm) || issn.includes(searchTerm);
    });
}
```
